<a|user|>\n${msg.content}<|end|>\n`;
        } else if (msg.role === 'assistant') {
          prompt += `<|assistant|>\n${msg.content}<|end|>\n`;
        }
      }
      prompt += '<|assistant|>\n';
      return prompt;
    }
    
    case 'generic':
    default: {
      // Simple generic format
      let prompt = '';
      for (const msg of messages) {
        prompt += `${msg.role}: ${msg.content}\n`;
      }
      prompt += 'assistant: ';
      return prompt;
    }
  }
}
