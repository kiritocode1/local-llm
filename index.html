Quick Start

// Import and create LLM
import { createLLM } from 'local-llm';

const llm = await createLLM({
  onLoadProgress: (p) => console.log(p.progress + '%')
});

// Chat with streaming
await llm.stream('Hello!', (token) => {
  console.log(token);
});

// Or attach to DOM elements
llm.attachToInput('#input', '#output');
